1. Factorial Calculation in JavaScript

<!DOCTYPE html>
<html>
<head>
    <title>Factorial of a Number</title>
</head>
<body>
    <script>
        var n = parseInt(prompt("Enter a number:")), factorial = 1;
        if (n < 0) {
            document.write("Factorial is not defined for negative numbers.");
        } else {
            for (let i = 1; i <= n; i++) {
                factorial *= i;
            }
            document.write("Factorial of " + n + " is " + factorial);
        }
    </script>
</body>
</html>
