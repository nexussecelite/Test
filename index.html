1. Factorial Calculation in JavaScript

Code
Live Demo
// Factorial calculation
function calculateFactorial() {
  var n = parseInt(document.getElementById('factorialInput').value);
  var result = document.getElementById('factorialResult');
  
  if (isNaN(n)) {
    result.textContent = "Please enter a valid number.";
    return;
  }
  
  var factorial = 1;
  if (n < 0) {
    result.textContent = "Factorial is not defined for negative numbers.";
  } else {
    for (let i = 1; i <= n; i++) {
      factorial *= i;
    }
    result.textContent = "Factorial of " + n + " is " + factorial;
  }
}
