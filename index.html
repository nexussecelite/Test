Series Calculator

S = 1² + 2² + ... + n²

#include <iostream>
#include <conio.h>
using namespace std;

class Series {
private:
    int n;
    long sum;
    long calculateSum() {
        long result = 0;
        for (int i = 1; i <= n; ++i) {
            result += (long)i * i;
        }
        return result;
    }
public:
    void readN() {
        cout << "Enter the value of n: ";
        cin >> n;
    }
    void computeSum() {
        sum = calculateSum();
    }
    void displaySum() {
        cout << "Sum of the series: " << sum << endl;
    }
};

int main() {
    clrscr();
    Series s;
    s.readN();
    s.computeSum();
    s.displaySum();
    getch();
    return 0;
}

Stack Implementation

Push, Pop, Peek, and Traverse operations

class Stack:
    def __init__(self):
        self.items = []

    def is_empty(self):
        return len(self.items) == 0

    def push(self, item):
        self.items.append(item)
        print(f"Pushed: {item}")

    def pop(self):
        if self.is_empty():
            print("Stack is empty. Cannot pop.")
            return None
        item = self.items.pop()
        print(f"Popped: {item}")
        return item

    def peek(self):
        if self.is_empty():
            print("Stack is empty. No top item.")
            return None
        return self.items[-1]

    def traverse(self):
        if self.is_empty():
            print("Stack is empty.")
            return
        print("Stack contents:")
        for item in reversed(self.items):
            print(item)

if __name__ == "__main__":
    stack = Stack()
    stack.push(90)
    stack.push(80)
    stack.push(70)
    stack.traverse()
    stack.pop()
    stack.traverse()
    stack.pop()
    stack.pop()
    stack.traverse()

Pattern Search

Naive string matching algorithm

def naive_pattern_search(text, pattern):
    n = len(text)
    m = len(pattern)
    for i in range(n - m + 1):
        j = 0
        while j < m:
            if text[i + j] != pattern[j]:
                break
            j += 1
        if j == m:
            print(f"Pattern found at index {i}")

# Example usage
text = "AABAACAADAABAABA"
pattern = "AABA"
naive_pattern_search(text, pattern)

Prime Number Check

Determine if a number is prime

#include <iostream>
#include <conio.h>
using namespace std;

int main() {
    int num, i;
    clrscr();
    cout << "Enter a positive integer: ";
    cin >> num;
    if (num <= 1) {
        cout << num << " is not a prime number." << endl;
        getch();
        return 0;
    }
    for (i = 2; i <= num / 2; ++i) {
        if (num % i == 0) {
            cout << num << " is not a prime number." << endl;
            getch();
            return 0;
        }
    }
    cout << num << " is a prime number." << endl;
    getch();
    return 0;
}

Sorting Algorithms

Bubble, Selection, and Insertion Sort

# Bubble Sort
def bubble_sort(arr):
    n = len(arr)
    for i in range(n):
        swapped = False
        for j in range(0, n - i - 1):
            if arr[j] > arr[j + 1]:
                arr[j], arr[j + 1] = arr[j + 1], arr[j]
                swapped = True
        if not swapped:
            break
    return arr

# Selection Sort
def selection_sort(arr):
    n = len(arr)
    for i in range(n):
        min_index = i
        for j in range(i + 1, n):
            if arr[j] < arr[min_index]:
                min_index = j
        arr[i], arr[min_index] = arr[min_index], arr[i]
    return arr

# Insertion Sort
def insertion_sort(arr):
    for i in range(1, len(arr)):
        key = arr[i]
        j = i - 1
        while j >= 0 and arr[j] > key:
            arr[j + 1] = arr[j]
            j -= 1
        arr[j + 1] = key
    return arr

# Example usage
arr = [64, 25, 12, 22, 11]
print("Bubble Sort Result:", bubble_sort(arr.copy()))
print("Selection Sort Result:", selection_sort(arr.copy()))
print("Insertion Sort Result:", insertion_sort(arr.copy()))

Merge Sort

Efficient divide-and-conquer algorithm

def merge_sort(arr):
    if len(arr) <= 1:
        return arr
    mid = len(arr) // 2
    left = merge_sort(arr[:mid])
    right = merge_sort(arr[mid:])
    return merge(left, right)

def merge(left, right):
    sorted_arr = []
    i = j = 0
    while i < len(left) and j < len(right):
        if left[i] < right[j]:
            sorted_arr.append(left[i])
            i += 1
        else:
            sorted_arr.append(right[j])
            j += 1
    sorted_arr.extend(left[i:])
    sorted_arr.extend(right[j:])
    return sorted_arr

# Example usage
arr = [38, 27, 43, 3, 9, 82, 10]
print("Original Array:", arr)
sorted_arr = merge_sort(arr)
print("Sorted Array:", sorted_arr)
