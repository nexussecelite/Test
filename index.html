1. Factorial Calculation in JavaScript

Code
Live Demo
<!DOCTYPE html>
<html>
<head>
  <title>Factorial Calculator</title>
  <script>
    function calculateFactorial() {
      var n = parseInt(document.getElementById('factorialInput').value);
      var result = document.getElementById('factorialResult');
      
      if (isNaN(n)) {
        result.textContent = "Please enter a valid number.";
        return;
      }
      
      var factorial = 1;
      if (n < 0) {
        result.textContent = "Factorial is not defined for negative numbers.";
      } else {
        for (let i = 1; i <= n; i++) {
          factorial *= i;
        }
        result.textContent = "Factorial of " + n + " is " + factorial;
      }
    }
  </script>
</head>
<body>
  <div class="calculator">
    <h1>Factorial Calculator</h1>
    <div>
      <input type="number" id="factorialInput" placeholder="Enter a number">
      <button onclick="calculateFactorial()">Calculate</button>
    </div>
    <div id="factorialResult" class="result"></div>
  </div>
</body>
</html>

4. Simple Password Check

Code
Live Demo
<!DOCTYPE html>
<html>
<head>
    <title>Password Check</title>
    <script type="text/javascript">
        function chkPass() {
            var first = document.frm.pass1.value;
            var second = document.frm.pass2.value;

            if (first == second) {
                return true;
            } else {
                alert("Both Passwords should be the Same");
                return false;
            }
        }
    </script>
</head>
<body>
    <form name="frm" action="#" onsubmit="return chkPass()">
        <font size="6">
            Enter Password: <input type="password" name="pass1"/><br>
            Confirm Password: <input type="password" name="pass2"/><br>
            <input type="submit" value="Submit"/>
        </font>
    </form>
</body>
